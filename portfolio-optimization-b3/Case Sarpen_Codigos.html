# ---------- 2. Volatilidade anualizada ----------
# desvio padrão diário × sqrt(252)
vol_anualizada = carteira_eq.std() * np.sqrt(dias_ano)

# ---------- 3. Sharpe Ratio ----------
# risco diário = CDI
rf_diario = df['CDI']

# excesso de retorno diário da carteira
excesso_diario = carteira_eq - rf_diario

sharpe = (excesso_diario.mean() * dias_ano) / (carteira_eq.std() * np.sqrt(dias_ano))

# ---------- 4. Máximo Drawdown ----------
# curva acumulada
curva = (1 + carteira_eq).cumprod()

# máxima marcação de topo ao longo do tempo
rolling_max = curva.cummax()

# drawdown = quanto caiu a partir do topo
drawdown = (curva / rolling_max) - 1

max_drawdown = drawdown.min()

# ---------- 5. Beta contra IBOV ----------
# beta = cov(carteira, ibov) / var(ibov)
cov = np.cov(carteira_eq.dropna(), df['IBOV'].dropna())[0][1]
var_ibov = df['IBOV'].var()

beta = cov / var_ibov

# 5) MÉTRICAS — Low-Vol (OOS) vs Equal-Weighted (FULL) vs IBOV (FULL)

import numpy as np
import pandas as pd

dias_ano = 252

# ----------------------------
# Função geral de métricas
# ----------------------------
def metrics_from_returns(ret_series, rf_series=None, dias_ano=252):
    s = ret_series.dropna()
    if rf_series is None:
        rf = pd.Series(0.0, index=s.index)
    else:
        rf = rf_series.reindex(s.index).ffill().fillna(0.0)

    ret_ann = s.mean() * dias_ano
    vol_ann = s.std() * np.sqrt(dias_ano)

    excesso = s - rf
    sharpe = (excesso.mean() / s.std()) * np.sqrt(dias_ano) if s.std() != 0 else np.nan

    cum = (1 + s).cumprod()
    rolling_max = cum.cummax()
    drawdown = cum / rolling_max - 1
    max_dd = drawdown.min()

    return {
        "Retorno Anual (%)": ret_ann * 100,
        "Vol (%)": vol_ann * 100,
        "Sharpe": sharpe,
        "MaxDD (%)": max_dd * 100
    }


# ================================
# 5.1 — Low-Vol OOS
# ================================
ret_lowvol_oos = ret_lowvol[oos_mask]
rf_lowvol = df["CDI"].reindex(ret_lowvol_oos.index)
ibov_lowvol = df["IBOV"].reindex(ret_lowvol_oos.index)

metrics_lowvol = metrics_from_returns(ret_lowvol_oos, rf_lowvol)

# Beta Low-Vol
cov_lv_ibov = np.cov(ret_lowvol_oos.dropna(), ibov_lowvol.dropna())[0,1]
beta_lowvol = cov_lv_ibov / ibov_lowvol.var()
metrics_lowvol["Beta"] = beta_lowvol


# ================================

# ================================
ret_eq_full = carteira_eq                # já calculado no problema 1
rf_full = df["CDI"]

metrics_eq = metrics_from_returns(ret_eq_full, rf_full)

cov_eq_ibov = np.cov(ret_eq_full.dropna(), df["IBOV"].dropna())[0,1]
beta_eq = cov_eq_ibov / df["IBOV"].var()
metrics_eq["Beta"] = beta_eq


# ================================
# 5.3 — IBOV FULL SAMPLE

metrics_ibov = metrics_from_returns(df["IBOV"], df["CDI"])

# Função oficial de métricas para todo o notebook
# ============================================================
def metrics(ret, rf, benchmark=None):
    ret = ret.dropna()
    rf = rf.reindex(ret.index).fillna(0)

    # Retorno anualizado
    ret_annual = ret.mean() * 252

    # Volatilidade anualizada
    vol_annual = ret.std() * np.sqrt(252)

    # Sharpe
    sharpe = ((ret - rf).mean() / ret.std()) * np.sqrt(252)

    # Max Drawdown
    cum = (1 + ret).cumprod()
    rollmax = cum.cummax()
    mdd = (cum / rollmax - 1).min()

    # Beta (se houver benchmark)
    if benchmark is not None:
        bench = benchmark.reindex(ret.index).dropna()
        aligned = ret.loc[bench.index]
        beta = np.cov(aligned, bench)[0,1] / bench.var()
    else:
        beta = np.nan

    return {
        "Retorno Anualizado (%)": ret_annual * 100,
        "Volatilidade (%)": vol_annual * 100,
        "Sharpe": sharpe,
        "Máx Drawdown (%)": mdd * 100,
        "Beta vs IBOV": beta
    }

# 2) TESTE DE SHARPE - LEDOIT & WOLF (2008)
# ===============================
import numpy as np

def ledoit_wolf_sharpe_test(r1, r2):
    """
    Teste de Ledoit-Wolf para diferença de Sharpe.
    Retorna estatística Z e p-value aproximado (two-sided).
    """
    r1 = r1.dropna()
    r2 = r2.reindex(r1.index).dropna()
    n = len(r1)
    
    # Sharpe individual
    s1 = r1.mean() / r1.std()
    s2 = r2.mean() / r2.std()
    
    # Diferença de Sharpe
    d = s1 - s2
    
    # Covariâncias necessárias
    cov11 = np.mean((r1 - r1.mean())**2)
    cov22 = np.mean((r2 - r2.mean())**2)
    cov12 = np.mean((r1 - r1.mean())*(r2 - r2.mean()))
    
    # Estimador Ledoit-Wolf
    var_d = (cov11 / (r1.std()**2) +
             cov22 / (r2.std()**2) -
             2 * cov12 / (r1.std() * r2.std())) / n
    
    z_stat = d / np.sqrt(var_d)
    
    # p-value two-sided
    from scipy.stats import norm
    p_value = 2 * (1 - norm.cdf(abs(z_stat)))
    
    return d, z_stat, p_value, s1, s2

print("===== LEDOIT-WOLF: Low-Vol vs Equal-Weighted =====")
